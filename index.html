Installation

<!DOCTYPE html>
<html>
<head>
	<!-- Latest compiled and minified Bootstrap CSS -->
	<link rel="stylesheet" href="//netdna.bootstrapcdn.com/bootstrap/3.2.0/css/bootstrap.min.css">

	<!-- Optional Bootstrap theme -->
	<link rel="stylesheet" href="//netdna.bootstrapcdn.com/bootstrap/3.2.0/css/bootstrap-theme.min.css">

	<link rel="stylesheet" href="dist/css/bootstrap-submenu.min.css">

	<!-- jQuery (necessary for Bootstrap-submenu (JavaScript) plugin) -->
	<script src="https://code.jquery.com/jquery.min.js" defer></script>
	<script src="dist/js/bootstrap-submenu.js" defer></script>
</head>
<body>
  ...
</body>
</html>

Enable Bootstrap-submenu via JavaScript:

$('.dropdown-submenu > a').submenupicker();
